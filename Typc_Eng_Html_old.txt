		});

<script>
function openNav() {
	$('#mySidenav').css('display','block');
}

/* Set the width of the side navigation to 0 */
function closeNav() {
	$('#mySidenav').css('display','none');
	if($(window).width() > 768){
		$('.side-menu-btn').focus();
	}
}

$(".SideNavTitle a").last().focusout(function(){
	$('#mySidenav').css('display','none');
});


$("#sitemaplinkx").click(function(){
	chkText = $("#webMapArea").css("display");
	$("#webMapArea").slideToggle("slow");
	if (chkText == "none"){
		$("#webMap").find("button").first().html("<i class='fas fa-chevron-up'></i>");
		$(this).attr("title","Close Site Navigation")
	}else{
		$("#webMap").find("button").first().html("<i class='fas fa-chevron-down'></i>");
		$(this).attr("title","Open Site Navigation")
	}
	$("a#sitemaplinkx").focus();
});


// Scroll to top btn
if($('.scroll-to-top').length){
	//Check to see if the window is top if not then display button
	$(window).scroll(function(){
		if ($(this).scrollTop() > 250) {
		$('.scroll-to-top').fadeIn();
		} else {
		$('.scroll-to-top').fadeOut();
		}
	});

	//Click event to scroll to top
	$('.scroll-to-top').on('click', function(){
		$('html, body').animate({scrollTop : 0},800);
		return false;
	});
}
